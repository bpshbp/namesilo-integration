const express = require("express"); const axios = require("axios"); const cors = require("cors"); const Stripe = require("stripe"); // Configuration const app = express(); const PORT = 5000; const NAMESILO_API_KEY = "090962cfcbf6da94f133"; const STRIPE_SECRET_KEY = "your_stripe_secret_key"; // Replace with your actual Stripe secret key const stripe = Stripe(STRIPE_SECRET_KEY); app.use(cors()); app.use(express.json()); app.use(express.static("public")); // Serve the frontend // Route to check domain availability app.get("/api/check-domain", async (req, res) => { const { domain } = req.query; if (!domain) { return res.status(400).json({ error: "Domain name is required" }); } try { const url = `https://www.namesilo.com/api/checkRegisterAvailability?key=${NAMESILO_API_KEY}&domain=${domain}&type=xml`; const response = await axios.get(url); res.status(200).send(response.data); } catch (error) { res.status(500).json({ error: "Failed to check domain availability" }); } }); // Route to create a Stripe payment session app.post("/api/create-checkout-session", async (req, res) => { const { plan } = req.body; const plans = { basic: { name: "Basic Hosting", price: 500 }, standard: { name: "Standard Hosting", price: 1000 }, premium: { name: "Premium Hosting", price: 2000 }, }; const selectedPlan = plans[plan]; if (!selectedPlan) { return res.status(400).json({ error: "Invalid plan selected" }); } try { const session = await stripe.checkout.sessions.create({ payment_method_types: ["card"], line_items: [ { price_data: { currency: "usd", product_data: { name: selectedPlan.name, }, unit_amount: selectedPlan.price, }, quantity: 1, }, ], mode: "payment", success_url: "http://localhost:5000/success.html", cancel_url: "http://localhost:5000/cancel.html", }); res.json({ url: session.url }); } catch (error) { res.status(500).json({ error: "Failed to create payment session" }); } }); app.listen(PORT, () => { console.log(`Server running on http://localhost:${PORT}`); });
